# Column definitions

```tsx
interface StyleProps {
  width: number;
  minWidth: number;
  maxWidth: number;
  hidden: boolean;
  flex: number;
  menu?: boolean | Partial<MenuProps>;
}
export interface MenuProps {
  column: boolean;
  filter: boolean;
  grid: boolean;
}
interface ColumnDef extends Partial<StyleProps> {
  headerName: string;
  field: string;
  children?: ColumnDef[];
  formatter?: (value: unknown) => string;

}
```

# Column definition example

```tsx
const columns: ColumnDef[] = [
  {
    headerName: 'Name',
    field: 'name',
    minWidth: 100,
    maxWidth: 200,
    menu: {
      column: true,
      filter: true,
      grid: true,
    },
    formatter: (value) => value.toUpperCase(),
  },
  {
    headerName: 'Age',
    field: 'age',
    width: 100,
    menu: {
      column: true
    }
  },
  {
    headerName: 'Address',
    field: 'address',
    menu: true
  },
  {
    headerName: 'Children',
    field: 'children',
    children: [
      {
        headerName: 'Name',
        field: 'name',
      },
      {
        headerName: 'Age',
        field: 'age',
      },
    ],
  },
];
```

# Properties

## `headerName`
### Type: `string`
### Required: `true`

The name of the column.

## `field`
### Type: `string`
### Required: `true`

The field to be displayed in the column. Must match the field name in the data.

## `children`
### Type: `ColumnDef[]`
### Required: `false`
 
The children columns of the column.

## `formatter`
### Type: `(value: unknown) => string`
### Required: `false`

A function to format the value of the column.

## `width`
### Type: `number`
### Required: `false`

The width of the column.

## `minWidth`
### Type: `number`
### Required: `false`

The minimum width of the column.

## `maxWidth`
### Type: `number`
### Required: `false`

The maximum width of the column.

## `hidden`
### Type: `boolean`
### Required: `false`

Whether the column is hidden.

## `flex`
### Type: `number`
### Required: `false`

The flex value of the column. Follows the same rules as the flex property in CSS.

## `menu`
### Type: `boolean | Partial<MenuProps>`
### Required: `false`

Whether to show the menu for the column. If `true`, it will show the default menu. If an object, it will show the menu with the properties specified in the object.
